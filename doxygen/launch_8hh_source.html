87  // TIP: fold-expression allows different types at each level
88  template<class U>
89  auto & operator|(const launch_combine<U> & c) const {
90  if constexpr(std::is_same_v<T, std::nullptr_t>)
91  return c;
92  else {
93  if constexpr(!std::is_same_v<U, std::nullptr_t>) {
94  static_assert(std::is_same_v<T, U>, "implied launch types conflict");
95  if constexpr(!std::is_same_v<T, std::monostate>)
96  flog_assert(t == c.t,

98  }
99  return *this;
100  }
101  }

104  return std::monostate();
