20  struct policy {

22  policy(std::size_t m = 0, // minimum total slots
23  std::size_t e = 0, // minimum extra slots
24  float l = 0, // lowest fill fraction allowed
25  float h = 1, // highest
26  float s = 0) // speed control on [0,1]

29  std::size_t operator()(std::size_t n, std::size_t cap) const {
30  // n on [floor(lo*c),hi*c] preserves cap (although we have no mechanism
31  // for a task to indicate to its caller the choice not to change it).
32  const auto div = [n](float d) -> std::size_t {
33  return std::nearbyint((n + .5f) / d);
34  };
35  return std::max(min,
36  std::max(n + extra,
37  n > hi * cap ? div(lo * slow)
38  : n >= std::size_t(lo * cap) ? cap : div(hi / slow)));
39  }
