214 one_to_allv(F const & f, MPI_Comm comm = MPI_COMM_WORLD) {
215  using return_type = std::decay_t<decltype(f(1, 1))>;
216 
217  auto [rank, size] = info(comm);

224  for(int r = 1; r < size; ++r)
225  send.push_back(f(r, size));
226  test(MPI_Scatter(MPI_IN_PLACE,
227  0,
228  MPI_DATATYPE_NULL,
229  send.data(),

231  type<return_type>(),
232  comm));
233  if(rank)
234  return send.front();
235  }
236  else {
237  detail::vector v(size);
238  v.skip(); // v.sz used even off-root
239  if(rank == 0) {
240  for(int r = 1; r < size; ++r)
241  v.put(f(r, size));
242  }
243 
244  test(MPI_Scatter(v.sz.data(),
245  1,
246  MPI_INT,
247  rank ? v.sz.data() : MPI_IN_PLACE,
248  1,
249  MPI_INT,
250  0,
251  comm));
252  if(rank)
253  v.data.resize(v.sz.front());

262  comm));

355 auto
356 one_to_alli(F && f, std::size_t mem, MPI_Comm comm = MPI_COMM_WORLD) {
357  using R = std::decay_t<decltype(f(1, 1))>;
358  using M = std::conditional_t<bit_copyable_v<R>,
359  detail::bit_message<R>,
360  detail::serial_message<R>>;
361 
362  const auto [rank, size] = info(comm);
363  if(!rank) {
364  {
365  const auto n = size - 1;
366  std::vector<MPI_Request> req;
367  std::vector<MPI_Status> stat(n);
368  {
369  std::vector<M> val; // parallel to req
370  std::vector<int> done(n);
371  std::stack<int> free; // inactive requests
372  std::size_t used = 0;
373  for(int r = 1; r < size; ++r) {
374  const int i = [&] {
375  if(free.empty()) {
376  // Capturing a structured binding requires C++20:
377  val.emplace_back(f, r, n + 1);
378  return int(req.size()); // created below
379  }
380  else {
381  const auto ret = free.top();
382  free.pop();
383  val[ret] = {f, r, n + 1};
384  return ret;
385  }
386  }();
387  const auto & v = val[i];
388  if(used && used + v.bytes() > mem) {
389  int count;
390  test(MPI_Waitsome(
391  req.size(), req.data(), &count, done.data(), stat.data()));
392  // Clear data for completed sends:
393  for(auto j : util::span(done).first(count)) {
394  auto & w = val[j];
395  used -= w.bytes();
396  w.reset();
397  free.push(j);
398  }
399  }
400  if(i == int(req.size()))
401  req.emplace_back();
402  // Discourage buffering (at the cost of needless synchronization):
403  test(MPI_Issend(v.data(), v.count(), M::type(), r, 0, comm, &req[i]));
404  used += v.bytes();
405  }
406  }
407  test(MPI_Waitall(req.size(), req.data(), stat.data()));
408  }
409  return f(0, size);
410  }
411  else {
412  M ret(0, 0, comm);
413  test(MPI_Recv(
414  ret.data(), ret.count(), M::type(), 0, 0, comm, MPI_STATUS_IGNORE));
415  return std::move(ret).get();
416  }
417 }

436 all_to_allv(F const & f, MPI_Comm comm = MPI_COMM_WORLD) {
437  using return_type = std::decay_t<decltype(f(1, 1))>;
438 
439  auto [rank, size] = info(comm);
440  std::vector<return_type> result;
441  result.reserve(size);

443  for(int r = 0; r < size; ++r)
444  result.push_back(f(r, size));
445  test(MPI_Alltoall(MPI_IN_PLACE,
446  0,
447  MPI_DATATYPE_NULL,
448  result.data(),
449  1,
450  type<return_type>(),
451  comm));
452  }
453  else {
454  detail::vector recv(size);
455  {
456  detail::vector send(size);
457 
458  for(int r = 0; r < size; ++r) {
459  if(r == rank)
460  send.skip();
461  else
462  send.put(f(r, size));
463  } // for
464 
465  recv.sz.resize(size);
466  test(MPI_Alltoall(
467  send.sz.data(), 1, MPI_INT, recv.sz.data(), 1, MPI_INT, comm));
468 
469  {
470  int o = 0;
471  for(const auto n : recv.sz) {
472  recv.off.push_back(o);
473  o += n;
474  }
475  recv.data.resize(o);
476  }
477 
478  test(MPI_Alltoallv(send.data.data(),
479  send.sz.data(),
480  send.off.data(),
481  MPI_BYTE,

484  recv.off.data(),

487  }

495  } // for
496  }

499 } // all_to_allv
