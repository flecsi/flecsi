291  template<index_space S, axis A, range SE>
292  std::size_t size() {
293  static_assert(
294  std::size_t(SE) < hypercubes::size, "invalid size identifier");
295  if constexpr(SE == range::logical) {
296  return logical<S, 1, A>() - logical<S, 0, A>();
297  }
298  else if constexpr(SE == range::extended) {
299  return extended<S, 1, A>() - extended<S, 0, A>();
300  }
301  else if constexpr(SE == range::all) {
302  return extents<S, A>();
303  }
304  else if constexpr(SE == range::boundary_low) {
305  return logical<S, 0, A>() - extended<S, 0, A>();
306  }
307  else if constexpr(SE == range::boundary_high) {
308  return extended<S, 1, A>() - logical<S, 1, A>();
309  }
310  else if constexpr(SE == range::ghost_low) {
311  if(!is_low<S, A>())
312  return logical<S, 0, A>();
313  else
314  return 0;
315  }
316  else if constexpr(SE == range::ghost_high) {
317  if(!is_high<S, A>())
318  return extents<S, A>() - logical<S, 1, A>();
319  else
320  return 0;
321  }
322  else if constexpr(SE == range::global) {
323  return global<S, A>();
324  }
325  }

327  template<index_space S, axis A, range SE>
328  auto extents() {
329  static_assert(
330  std::size_t(SE) < hypercubes::size, "invalid extents identifier");
331 
332  if constexpr(SE == range::logical) {
333  return make_ids<S>(
334  util::iota_view<util::id>(logical<S, 0, A>(), logical<S, 1, A>()));
335  }
336  else if constexpr(SE == range::extended) {
337  return make_ids<S>(
338  util::iota_view<util::id>(extended<S, 0, A>(), extended<S, 1, A>()));
339  }
340  else if constexpr(SE == range::all) {
341  return make_ids<S>(util::iota_view<util::id>(0, extents<S, A>()));
342  }
343  else if constexpr(SE == range::boundary_low) {
344  return make_ids<S>(util::iota_view<util::id>(0, size<S, A, SE>()));
345  }
346  else if constexpr(SE == range::boundary_high) {
347  return make_ids<S>(util::iota_view<util::id>(
348  logical<S, 1, A>(), logical<S, 1, A>() + size<S, A, SE>()));
349  }
350  else if constexpr(SE == range::ghost_low) {
351  return make_ids<S>(util::iota_view<util::id>(0, size<S, A, SE>()));
352  }
353  else if constexpr(SE == range::ghost_high) {
354  return make_ids<S>(util::iota_view<util::id>(
355  logical<S, 1, A>(), logical<S, 1, A>() + size<S, A, SE>()));
356  }
357  else {
358  flog_error("invalid range");
359  }
360  }
