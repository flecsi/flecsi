264  // Utility to transfer the contents of ragged rows via buffers.
265  struct ragged {
266  explicit ragged(Buffer & b) : skip(b.off), w(b) {}
267 
268  template<class R> // accessor or mutator
269  bool operator()(const R & rag, std::size_t i) {
270  const auto row = rag[i];
271  const auto n = row.size();
272  auto & b = w.get_buffer();
273  if(skip < n) {
274  // Each row's record is its index, the number of elements remaining to
275  // write in it (which might not all fit), and then the elements.
276  if(!w(i) || !w(n - skip))
277  return false;
278  for(auto s = std::exchange(skip, 0); s < n; ++s)
279  if(w(row[s]))
280  ++b.off;
281  else {
282  flog_assert(b.len > 2, "no data fits");
283  return false;
284  }
285  }
286  else
287  skip -= n;
288  return true;
289  }
290 
291  // For the first use in each communication:
292  static ragged truncate(Buffer & b) {
293  b.off = 0;
294  return ragged(b);
295  }

297  template<class R, class F> // F: remote/shared index -> local/ghost index
298  static void read(const R & rag, const Buffer & b, F && f) {
299  for(Buffer::reader r{&b}; r;) {
300  const auto row = rag[f(r.get<std::size_t>())];
301  if(!r)
302  break; // in case the write stopped mid-record
303  for(std::size_t n = r(); r && n--;)
