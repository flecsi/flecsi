31 inline std::vector<std::size_t>
32 factor(std::size_t np) {
33  std::vector<std::size_t> facs;
34  const auto test = [&](std::size_t f) {
35  while(!(np % f)) {
36  facs.push_back(f);
37  np /= f;
38  }
39  };
40 
41  // Trivial wheel factorization:
42  test(2);
43  test(3);
44  for(std::size_t i = 5, step = 2; i * i <= np; i += step, step = 2 * 3 - step)
45  test(i);
46  if(np > 1) // i.e., the largest prime factor is >3 and unique
47  facs.push_back(np);
48  std::reverse(facs.begin(), facs.end());
49 
50  return facs;
51 } // factor

57 inline narray_impl::colors
58 distribute(Color np, std::vector<std::size_t> indices) {
59  narray_impl::colors parts(indices.size(), 1);
60 
61  auto facs = factor(np);
62 
63  // greedy decomposition
64  for(auto fac : facs) {
65  auto maxind = std::distance(
66  indices.begin(), std::max_element(indices.begin(), indices.end()));
67  parts[maxind] *= fac;
68  indices[maxind] /= fac;
69  }
70 
71  return parts;
72 } // decomp

82 inline auto
83 orientation(Dimension dimension,
84  const narray_impl::colors & color_indices,
85  const narray_impl::colors & axis_colors) {
86  using namespace narray_impl;
87 
88 #define FACE(e, c, c0, c1, c2) \
89  ((((e) == 0) && ((e) == ((c)-1))) \
90  ? c1 | c2 \
91  : ((0 < (e) && e < ((c)-1)) ? c0 : ((e) == 0 ? c1 : c2)))
92 
93  std::uint32_t faces = interior;
94  std::uint32_t shft{low};
95  for(Dimension axis = 0; axis < dimension; ++axis) {
96  faces |=
97  FACE(color_indices[axis], axis_colors[axis], interior, shft, shft << 1);
98  shft <<= 2;
99  } // for
100 #undef FACE
101 
102  return faces;
103 } // orientation

125  idxco.logical[1].resize(dimension);

131  std::pair<std::size_t, /* owner color */

139  idxco.global[axis] = axcm[axis].indices();
140  idxco.extents[axis] = axcm[axis].indices(axis_color, 0);
141  idxco.offset[axis] = axcm[axis].index_offset(color_indices[axis], 0);
142  idxco.logical[0][axis] = 0;
143  idxco.logical[1][axis] = axcm[axis].indices(axis_color, 0);

145  std::uint32_t bits = faces >> axis * 2;

150  color layer. Therefore, we do not add halo extensions.
151  */

164  */

169  idxco.logical[0][axis] += bdepths[axis];
170  idxco.logical[1][axis] += bdepths[axis];
171 
172  idxco.extended[0][axis] = 0;
173  idxco.extended[1][axis] = idxco.logical[1][axis];
174 
175  ghstitvls[axis].push_back({color_indices[axis] + 1,
176  {idxco.logical[1][axis], idxco.logical[1][axis] + hdepths[axis]}});
177 
178  if(periodic[axis]) {
179  flog_assert(bdepths[axis] > 0,
180  "periodic boundaries require non-zero boundary depth");
181

183  {idxco.extended[0][axis], idxco.extended[0][axis] + bdepths[axis]}});
184  } // if
185  }
186  else if(bits & high) {
187  /*
188  This dimension is a high edge.
189  */
190

192 
193  idxco.offset[axis] -= hdepths[axis];
194 
195  idxco.logical[0][axis] += hdepths[axis];
196  idxco.logical[1][axis] += hdepths[axis];
197 
198  idxco.extended[0][axis] = idxco.logical[0][axis];
199  idxco.extended[1][axis] = idxco.logical[1][axis] + bdepths[axis];
200 
201  ghstitvls[axis].push_back({color_indices[axis] - 1,
202  {idxco.logical[0][axis] - hdepths[axis], idxco.logical[0][axis]}});
203 
204  if(periodic[axis]) {
205  flog_assert(bdepths[axis] > 0,
206  "periodic boundaries require non-zero boundary depth");
207 
208  ghstitvls[axis].push_back({0,
209  {idxco.logical[1][axis], idxco.logical[1][axis] + bdepths[axis]}});
210  } // if
211  }
212  else {
213  /*
214  This dimension is interior.
215  */
216 
217  idxco.extents[axis] += 2 * hdepths[axis];
218 
219  idxco.offset[axis] -= hdepths[axis];
220 
221  idxco.logical[0][axis] += hdepths[axis];
222  idxco.logical[1][axis] += hdepths[axis];
223 
224  idxco.extended[0][axis] = idxco.logical[0][axis];
225  idxco.extended[1][axis] = idxco.logical[1][axis];
226 
227  ghstitvls[axis].push_back({color_indices[axis] + 1,
228  {idxco.logical[1][axis], idxco.logical[1][axis] + hdepths[axis]}});
229  ghstitvls[axis].push_back({color_indices[axis] - 1,
230  {idxco.logical[0][axis] - hdepths[axis], idxco.logical[0][axis]}});
231  } // if
232  } // for
233 
234  return std::make_pair(idxco, ghstitvls);

241 inline auto
242 color(std::vector<narray_impl::coloring_definition> const & index_spaces,
243  MPI_Comm comm = MPI_COMM_WORLD) {
244  using namespace narray_impl;
245 
246  flog_assert(index_spaces.size() != 0, "no index spaces defined");
247 
248  const Dimension dimension = index_spaces[0].axis_colors.size();
249 
250  flog_assert(dimension < 17,
251  "current implementation is limited to 16 dimensions (uint32_t)");
252 
253  Color nc = 1;
254  auto [rank, size] = util::mpi::info(comm);
255  std::vector<std::map<std::size_t, index_coloring>> colorings;
256 
257  for(std::size_t is{0}; is < index_spaces.size(); ++is) {
258  auto const & axis_colors = index_spaces[is].axis_colors;
259  auto const & axis_extents = index_spaces[is].axis_extents;
260  auto const & axis_hdepths = index_spaces[is].axis_hdepths;
261  auto const & axis_bdepths = index_spaces[is].axis_bdepths;
262  std::vector<bool> const & axis_periodic = index_spaces[is].axis_periodic;
263  bool const diagonals = index_spaces[is].diagonals;
264  bool const create_plan = index_spaces[is].create_plan;
265 
266  flog_assert(axis_colors.size() == axis_extents.size(),
267  "argument mismatch: sizes(" << axis_colors.size() << "vs. "
268  << axis_extents.size()
269  << ") must be consistent");
270  flog_assert(axis_colors.size() == dimension,
271  "size must match the intended dimension(" << dimension << ")");
272 
273  /*
274  Create a color map for each dimension. Because we are using a
275  tensor-product strategy below, we can use each map to define the
276  sub colors and offsets for an individual axis.
277  */
278 
279  Color idx_colors = 1;
280  std::size_t indices{1};
281  std::vector<util::color_map> axcm;
282  for(Dimension d = 0; d < dimension; ++d) {
283  if(is == 0) {
284  nc *= axis_colors[d];
285  }
286  idx_colors *= axis_colors[d];
287  indices *= axis_extents[d];
288  axcm.emplace_back(axis_colors[d], axis_colors[d], axis_extents[d]);
289  } // for
290 
291  flog_assert(
292  idx_colors == nc, "index spaces must have a consistent number of colors");
293 
294  /*
295  Create a color map for the total number of colors (product of axis
296  colors) to the number of processes.
297  */
298 
299  util::color_map cm(size, idx_colors, indices);
300 
301  /*
302  If this index space should do data exchange, populate the
303  intervals and points that define the copy plan. An empty
304  copy plan shouldn't result in too much inefficiency, and it
305  is not straightforward to skip index spaces during narray
306  allocation, i.e., to skip copy plan creation alltogether.
307  */
308 
309  std::map<std::size_t, index_coloring> coloring;
310  if(create_plan) {
311  /*
312  Create a coloring for each color on this process.
313  */
314 
315  for(Color c = 0; c < cm.colors(rank); ++c) {
316  /*
317  Convenience functions to map between colors and indices.
318  */
319 
320  const auto co2idx = [](Color co, const colors & szs) {
321  colors indices;
322  for(auto sz : szs) {
323  indices.emplace_back(co % sz);
324  co /= sz;
325  }
326  return indices;
327  };
328 
329  const auto idx2co = [](const auto & idx, const auto & szs) {
330  auto pr = szs[0];
331  decltype(pr) co = 0;
332  for(std::size_t i{0}; i < idx.size() - 1; ++i) {
333  co += idx[i + 1] * pr;
334  pr *= szs[i + 1];
335  }
336  return co + idx[0];
337  };
338 
339  auto color_indices = co2idx(cm.color_id(rank, c), axis_colors);
340 
341  /*
342  Find our orientation within the color space.
343  */
344 
345  uint32_t faces = orientation(dimension, color_indices, axis_colors);
346 
347  /*
348  Make the coloring information for our color.
349  */
350 
351  auto [idxco, ghstitvls] = make_color(dimension,
352  color_indices,
353  axcm,
354  faces,
355  axis_hdepths,
356  axis_bdepths,
357  axis_periodic);
358 
359  // This won't be necessary with c++20. In c++17 lambdas can't
360  // capture structured bindings.
361  auto & idx_coloring = idxco;
362  auto & ghost_intervals = ghstitvls;
363 
364  /*
365  Here we compose the intervals from each sub-dimension to form
366  the actual full-dimensional subregions. These define the coloring.
367  */
368 
369  std::function<std::vector<std::pair<colors, hypercube>>(
370  Dimension, Dimension)>
371  expand = [&idx_coloring,
372  &color_indices,
373  &ghost_intervals,
374  diagonals,
375  &expand](Dimension dim, Dimension top) {
376  std::vector<std::pair<colors, hypercube>> sregs;
377 
378  for(Dimension axis = 0; axis < dim; ++axis) {
379  if(sregs.size() && dim == top) {
380  /*
381  Expand the subregions from the lower dimensions.
382  */
383 
384  auto subs = std::move(sregs);
385  sregs.clear();
386  for(size_t off{idx_coloring.logical[0][axis]};
387  off < idx_coloring.logical[1][axis];
388  ++off) {
389  for(auto & s : subs) {
390  s.first[axis] = color_indices[axis];
391  s.second[0][axis] = off;
392  s.second[1][axis] = off + 1;
393  sregs.emplace_back(s);
394  } // for
395  } // for
396  } // if
397 
398  /*
399  Add the subregions for this dimension.
400  */
401 
402  for(auto i : ghost_intervals[axis]) {
403  colors co(top, 0);
404  coord start(top, 0);
405  coord end(top, 0);
406  for(Dimension a = 0; a < axis; ++a) {
407  co[a] = color_indices[a];
408  start[a] = idx_coloring.logical[0][a];
409  end[a] = idx_coloring.logical[1][a];
410  } // for
411 
412  co[axis] = i.first;
413  start[axis] = i.second.first;
414  end[axis] = i.second.second;
415  sregs.push_back({co, hypercube{start, end}});
416 
417  if(diagonals && axis > 0) {
418  /*
419  Recurse to pull up lower dimensions.
420  */
421 
422  auto ssubs = expand(dim - 1, top);
423 
424  /*
425  Add axis information from this dimension to new diagonals.
426  */
427 
428  for(auto ss : ssubs) {
429  ss.first[axis] = i.first;
430  ss.second[0][axis] = i.second.first;
431  ss.second[1][axis] = i.second.second;
432  sregs.emplace_back(ss);
433  } // for
434  } // if
435  } // for
436  } // for
437 
438  return sregs;

447  /*
448  Compute a remote index from a global coordinate.
449  */

451  auto rmtidx = [dimension, &axis_bdepths](
452  index_coloring const & idxco, coord const & gidx) {
453  coord result(dimension);
454  for(Dimension axis = 0; axis < dimension; ++axis) {
455  uint32_t bits = idxco.faces >> axis * 2;
456  if(bits & low) {
457  result[axis] = gidx[axis] + axis_bdepths[axis];
458  }
459  else {
460  result[axis] = gidx[axis] - idxco.offset[axis];
461  }
462  }
463  return result;
464  };

466  /*
467  Map a local coordinate to a global one.
468  */
469 
470  auto l2g = [dimension, faces, &axis_bdepths](
471  index_coloring const & idxco, coord const & idx) {
472  coord result(dimension);
473  for(Dimension axis = 0; axis < dimension; ++axis) {
474  uint32_t bits = faces >> axis * 2;
475  if(bits & low) {
476  if(idx[axis] < idxco.logical[0][axis]) {
477  /* periodic low */
478  result[axis] =
479  idxco.global[axis] - axis_bdepths[axis] + idx[axis];
480  }
481  else {
482  result[axis] = idx[axis] - axis_bdepths[axis];
483  }
484  }
485  else if(bits & high && idx[axis] >= idxco.logical[1][axis]) {
486  /* periodic high */
487  result[axis] = idx[axis] - idxco.logical[1][axis];
488  }
489  else {
490  result[axis] = idxco.offset[axis] + idx[axis];
491  }
492  }
493  return result;
494  };

498  closed on the start of the interval, and open on the end. This
499  function is used below to close the end, so that the interval
500  can be converted into a memory offset interval.
501  */
502 
503  auto op2cls = [dimension](coord const & idx) {
504  coord result(dimension);
505  for(Dimension axis = 0; axis < dimension; ++axis) {
506  result[axis] = idx[axis] - 1;

535  /*
536  The subregions are defined by hypercubes. These must be broken
537  up into contiguous intervals. This lambda recurses each subregion
538  to break up the volume into contiguous chunks.
539  */
540 
541  std::function<void(Color, hypercube, Dimension)> make =
542  [&idx_coloring, &idxmap, &op2cls, &idx2co, &l2g, &rmtidx, &make](
543  Color clr, hypercube const & subregion, Dimension axis) {
544  if(axis == 0) {
545  // Compute the local memory interval.
546  auto const start = idx2co(subregion[0], idx_coloring.extents);
547  auto const end =
548  idx2co(op2cls(subregion[1]), idx_coloring.extents);
549 
550  // The output intervals are closed on the start
551  // and open on the end, i.e., [start, end)
552 
553  idx_coloring.intervals.push_back({start, end + 1});
554 
555  /*
556  Loop through the local interval sizes, and add the remote
557  pointer offsets.
558  */
559 
560  auto const gidx = l2g(idx_coloring, subregion[0]);
561  auto const ridx = rmtidx(idxmap.at(clr), gidx);
562  auto rmtoff = idx2co(ridx, idxmap.at(clr).extents);
563 
564  for(std::size_t off{0}; off < (end + 1) - start; ++off) {
565  idx_coloring.points[clr].push_back(
566  {start + off, rmtoff + off});
567  } // for
568  }
569  else {
570  // Recurse ranges at this axis to create contiguous intervals.
571  for(std::size_t r = subregion[0][axis]; r < subregion[1][axis];
572  ++r) {
573  hypercube rct = subregion;
574  rct[0][axis] = r;
575  rct[1][axis] = r + 1;
576 
577  make(clr, rct, axis - 1);
578  } // for
579  } // if
580  };
581 
582  make(co, s.second, dimension - 1);
583  } // for
584 
585  coloring.emplace(c, std::move(idx_coloring));
586  } // for
587  } // if

590  } // for
591 
592  return std::make_pair(nc, colorings);
