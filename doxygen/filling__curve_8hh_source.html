240  hilbert_curve(const std::array<point_t, 2> & range,
241  const point_t & p,
242  const std::size_t depth) {
243  *this = filling_curve<DIM, T, hilbert_curve>::min();
244  assert(depth <= max_depth_);
245  std::array<int_t, dimension> coords;
246  const int_t max_val = (int_t(1) << (bits_ - 1) / dimension) - 1;
247 
248  // Convert the position to integer
249  for(Dimension i = 0; i < dimension; ++i) {
250  double min = range[0][i];
251  double scale = range[1][i] - min;
252  coords[i] = std::min(max_val,
253  static_cast<int_t>((p[i] - min) / scale * (int_t(1) << (max_depth_))));
254  }
255  // Handle 1D case
256  if(dimension == 1) {
257  assert(value_ & 1UL << max_depth_);
258  value_ |= coords[0] >> dimension;
259  value_ >>= (max_depth_ - depth);
260  return;
261  }
262  int_t mask = static_cast<int_t>(1) << (max_depth_);
263  for(int_t s = mask >> 1; s > 0; s >>= 1) {
264  std::array<int_t, dimension> bits;
265  for(Dimension j = 0; j < dimension; ++j) {
266  bits[j] = (s & coords[j]) > 0;
267  }
268  if(dimension == 2) {
269  value_ += s * s * ((3 * bits[0]) ^ bits[1]);
270  rotation2d(s, coords, bits);
271  }
272  if(dimension == 3) {
273  value_ += s * s * s * ((7 * bits[0]) ^ (3 * bits[1]) ^ bits[2]);
274  rotation3d(s, coords, bits);
275  }
276  }
277  // Then truncate the key to the depth
278  value_ >>= (max_depth_ - depth) * dimension;
279  }

282  void coordinates(const std::array<point_t, 2> & range, point_t & p) {
283  int_t key = value_;
284  std::array<int_t, dimension> coords;
285  coords.fill(int_t(0));
286 
287  int_t n = int_t(1) << (max_depth_); // Number of cells to an edge.
288  for(int_t mask = int_t(1); mask < n; mask <<= 1) {
289  std::array<int_t, dimension> bits = {};
290  if(dimension == 3) {
291  bits[0] = (key & 4) > 0;
292  bits[1] = ((key & 2) ^ bits[0]) > 0;
293  bits[2] = ((key & 1) ^ bits[0] ^ bits[1]) > 0;
294  rotation3d(mask, coords, bits);
295  coords[0] += bits[0] * mask;
296  coords[1] += bits[1] * mask;
297  coords[2] += bits[2] * mask;
298  }
299  if(dimension == 2) {
300  bits[0] = (key & 2) > 0;
301  bits[1] = ((key & 1) ^ bits[0]) > 0;
302  rotation2d(mask, coords, bits);
303  coords[0] += bits[0] * mask;
304  coords[1] += bits[1] * mask;
305  }
306  key >>= dimension;
307  }
308  assert(key == int_t(1));
309  for(Dimension j = 0; j < dimension; ++j) {
310  double min = range[0][j];
311  double scale = range[1][j] - min;
312  p[j] =
313  min + scale * static_cast<double>(coords[j]) / (int_t(1) << max_depth_);
314  } // for
315  }

468  morton_curve(const std::array<point_t, 2> & range,
469  const point_t & p,
470  const std::size_t depth) {
471  *this = filling_curve<DIM, T, morton_curve>::min();
472  assert(depth <= max_depth_);
473  std::array<int_t, dimension> coords;
474  const int_t max_val = (int_t(1) << (bits_ - 1) / dimension) - 1;
475  for(Dimension i = 0; i < dimension; ++i) {
476  double min = range[0][i];
477  double scale = range[1][i] - min;
478  coords[i] = std::min(max_val,
479  static_cast<int_t>(
480  (p[i] - min) / scale *
481  static_cast<double>((int_t(1) << (bits_ - 1) / dimension))));
482  } // for
483  std::size_t k = 0;
484  for(std::size_t i = max_depth_ - depth; i < max_depth_; ++i) {
485  for(Dimension j = 0; j < dimension; ++j) {
486  int_t bit = (coords[j] & int_t(1) << i) >> i;
487  value_ |= bit << (k * dimension + j);
488  } // for
489  ++k;
490  } // for
491  } // morton_curve

494  void coordinates(const std::array<point_t, 2> & range, point_t & p) {
495  std::array<int_t, dimension> coords;
496  coords.fill(int_t(0));
497  int_t id = value_;
498  std::size_t d = 0;
499  while(id >> dimension != int_t(0)) {
500  for(Dimension j = 0; j < dimension; ++j) {
501  coords[j] |= (((int_t(1) << j) & id) >> j) << d;
502  } // for
503  id >>= dimension;
504  ++d;
505  } // while
506  constexpr int_t m = (int_t(1) << max_depth_) - 1;
507  for(Dimension j = 0; j < dimension; ++j) {
508  double min = range[0][j];
509  double scale = range[1][j] - min;
510  coords[j] <<= max_depth_ - d;
511  p[j] = min + scale * static_cast<double>(coords[j]) / m;
512  } // for
513  } // coordinates

519  std::array<point_t, 2> range(const std::array<point_t, 2> & range) {
520  // The result range
521  std::array<point_t, 2> result;
522  result[0] = range[0];
523  result[1] = range[1];
524  // Copy the key
525  int_t root = filling_curve<DIM, T, morton_curve>::root().value_;
526  // Extract x,y and z
527  std::array<int_t, dimension> coords;
528  coords.fill(int_t(0));
529  int_t id = value_;
530  std::size_t d = 0;
531  while(id != root) {
532  for(Dimension j = 0; j < dimension; ++j) {
533  coords[j] |= (((int_t(1) << j) & id) >> j) << d;
534  }
535  id >>= dimension;
536  ++d;
537  }
538  for(Dimension i = 0; i < dimension; ++i) {
539  // apply the reduction
540  for(std::size_t j = d; j > 0; --j) {
541  double nu = (result[0][i] + result[1][i]) / 2.;
542  if(coords[i] & (int_t(1) << j - 1)) {
543  result[0][i] = nu;
544  }
545  else {
546  result[1][i] = nu;
547  } // if
548  } // for
549  } // for
550  return result;
551  } // range
