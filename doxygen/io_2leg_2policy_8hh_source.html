201  inline void checkpoint_data(const std::string & file_name, bool attach_flag) {
202  Legion::Runtime * runtime = Legion::Runtime::get_runtime();
203  Legion::Context ctx = Legion::Runtime::get_context();
204

208  util::serial_put(p, h.field_string_map);

211 
212  const auto task_id =
213  attach_flag ? exec::leg::task_id<checkpoint_task<W, true>, loc | inner>
214  : exec::leg::task_id<checkpoint_task<W, false>, loc | leaf>;
215 
216  Legion::IndexLauncher checkpoint_launcher(task_id,
217  launch_space,
218  Legion::TaskArgument((void *)(task_args.data()), task_args.size()),
219  Legion::ArgumentMap());
220 
221  int idx = 0;
222  for(auto & it : hdf5_region_vector) {
223  checkpoint_launcher.add_region_requirement(
224  Legion::RegionRequirement(it.logical_partition,
225  0 /*projection ID*/,
226  W ? READ_ONLY : WRITE_DISCARD,
227  EXCLUSIVE,
228  it.logical_region));
229 
230  for(auto & it : it.field_string_map) {
231  checkpoint_launcher.region_requirements[idx].add_field(it.first);
232  }
233  idx++;
234  }
235 
236  {
237  log::devel_guard guard(io_tag);
238  flog_devel(info) << "Start " << (W ? "checkpoint" : "recover") << " file "

240  << hdf5_region_vector.size() << std::endl;
241  }
242 
243  Legion::FutureMap fumap =
244  runtime->execute_index_space(ctx, checkpoint_launcher);
245  fumap.wait_all_results();
