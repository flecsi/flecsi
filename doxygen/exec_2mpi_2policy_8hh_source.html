39 // AA is what the user gives us when calling execute(), PP is what
40 // the user defined function/task expects. PP may not be the same
41 // as AA, for example, user pass a field_reference as an argument
42 // to execute() but the task expects an data accessor as its formal
43 // parameter. In this case replace_argument replaces a
44 // field_reference with an accessor. This is done through various
45 // specialization of the exec::detail::task_param<> template.
46 template<class... PP, class... AA>
47 auto
48 replace_arguments(std::tuple<PP...> * /* to deduce PP */, AA &&... aa) {
49  // Specify the template arguments explicitly to produce references to
50  // unchanged arguments.
51  return std::tuple<decltype(exec::replace_argument<PP>(std::forward<AA>(
52  aa)))...>(exec::replace_argument<PP>(std::forward<AA>(aa))...);
53 }

57 template<auto & F, class Reduction, TaskAttributes Attributes, typename... Args>
58 auto
59 reduce_internal(Args &&... args) {
60  using util::mpi::test;
61  using Traits = util::function_traits<decltype(F)>;
62  using R = typename Traits::return_type;
63 
64  // replace arguments in args, for example, field_reference -> accessor.
65  auto params = exec::detail::replace_arguments(
66  static_cast<typename Traits::arguments_type *>(nullptr),
67  std::forward<Args>(args)...);
68 
69  // TIP: param_buffers is an RAII type. We create an instance and give
70  // the object a reference to the parameters and name of the task. We then
71  // assign it to the `finalize` variable so it is not destroyed immediately.
72  // The object will be destroyed when reduce_internal() returns. The
73  // ~param_buffers() will then perform the necessary clean up on the
74  // parameters (mostly calling mutator.commit()).
75  auto task_name = util::symbol<F>();
76  auto finalize = param_buffers{params, task_name};
77 
78  // Now we have accessors, we need to bind the accessor to real memory
79  // for the data field. We also need to patch up default conversion
80  // from args to params, especially for the future<>.
81  prolog(params, args...);
82 
83  // Different kinds of task invocation with flecsi::execute():
84  // 1. domain_size is a std::monostate: a single task launch. On a single
85  // rank (0) of our choice, we apply F to ARGS. Given the return type R of
86  // F, we return a future<R, single>. The client can later call
87  // future<>.get() on any rank to get the result. The value returned by
88  // .get should be the same on all ranks, implying a Broadcast is needed.
89  // 2. Reduction is not void: a true reduction task.
90  // We apply F to ARGS and reduce the return values with the Reduction
91  // operation. We then put the reduced value into a future<R,
92  // launch_type::single> and return it. Again, the client could call
93  // .get() to get the (same) result on any rank, implying either an
94  // Allreduce or Reduce/Broadcast is needed.
95  // 3. Reduction is void: an index launch. We apply F to ARGS on every rank,
96  // each will return a value r_i. The value r_i will be put into the the
97  // slot i in the future<R, index>. The client can call future<R,
98  // index>.get(j) to get the return value on any rank j. This implies an
99  // Allgather is needed.
100  //
101  const auto ds =
102  launch_size<Attributes, decltype(params)>(std::forward<Args>(args)...);
103  util::annotation::rguard<util::annotation::execute_task_user> ann{task_name};
104  if constexpr(std::is_same_v<decltype(ds), const std::monostate>) {
105  const bool root = !flecsi::run::context::instance().process();
106  // single launch, only invoke the user task on the Root.
107  if constexpr(std::is_void_v<R>) {
108  // void return type, just invoke, no return value to broadcast
109  if(root) {
110  std::apply(F, std::move(params));
111  }
112  return future<void>{};
113  }
114  else {
115  // TODO: consider non-blocking Bcast, issue Bcast here and wait for
116  // completion at .wait()/.get().
117  // TODO: wrap this in std::async and use std::future for async execution
118  // and communication.
119  R ret{};
120  if(root) {
121  ret = std::apply(F, std::move(params));
122  }
123 
124  // Broadcast the result from root to the rest of ranks
125  test(MPI_Bcast(&ret, 1, flecsi::util::mpi::type<R>(), 0, MPI_COMM_WORLD));
126 
127  // return future<R, launch_type::single> where clients on every rank
128  // will get the same value when calling .get().
129  return future<R>{std::move(ret)};
130  }
131  }
132  else {
133  flog_assert(ds == run::context::instance().processes(),
134  "MPI backend supports only per-rank index launches");
135  // index launch (including "mpi task"), invoke the user task on all ranks.
136  if constexpr(!std::is_void_v<Reduction>) {
137  static_assert(!std::is_void_v<R>, "can not reduce results of void task");
138 
139  // A real reduce operation, every rank needs to be able to access the
140  // same result through future<R>::get().
141  // 1. Call the F, get the local return value
142  R ret = std::apply(F, std::move(params));
143  // 2. Reduce the local return values with the Reduction (using its
144  // corresponding MPI_Op created by register_reduction<>()).
145  // TODO: consider non-blocking Allreduce, issue Iallreduce here and wait
146  // for completion at .wait()/.get().
147  test(MPI_Allreduce(MPI_IN_PLACE,
148  &ret,
149  1,
150  flecsi::util::mpi::type<R>(),
151  flecsi::exec::fold::wrap<Reduction, R>::op,
152  MPI_COMM_WORLD));
153  // 3. Put the reduced value in a future<R, single> (since there is only
154  // one final value) and return it.
155  return future<R>{std::move(ret)};
156  }
157  else if constexpr(!std::is_void_v<R>)
158  // There is an Allgather happening in the constructor of future<R, index>
159  // where the results from ranks are redistributed such that clients on
160  // every rank i can get the return value of rank j by calling get(j).
161  return future<R, exec::launch_type_t::index>{
162  std::apply(F, std::move(params))};
163  else {
164  // index launch of void functions, e.g. printf("hello world");
165  std::apply(F, std::move(params));
166  return future<void, exec::launch_type_t::index>{};
167  }
168  }
