88 template<class T>
89 struct particle : particle_base {
90  particle(size_type s, size_type p, size_type n) : free{p, n}, skip(s) {}
91  // This class is indestructible; we run T's destructor when necessary.
92  template<class... AA>
93  link emplace(AA &&... aa) {
94  struct guard {
95  guard(particle & p) : p(p), ret(p.free) {}
96  ~guard() {
97  if(fail)
98  p.free = ret;
99  }
100  particle & p;
101  link ret;
102  bool fail = true;
103  } g(*this);
104  new(&data) T(std::forward<AA>(aa)...);
105  g.fail = false;
106  return g.ret;
107  }
108  void reset() noexcept {
109  data.~T();
110  }
111  void reset(size_type p, size_type n) noexcept {
112  reset();
113  free = {p, n};
114  }
115 
116  union
117  {
118  T data; // pointer-interconvertible if standard-layout
119  link free;
120  };
121  // To avoid a separate field allocation, this member has several meanings:
122  // If first element (never part of a run): index of head of free list
123  // Otherwise, if data exists: 0
124  // Otherwise, if first or last of its free run: length of the run
125  // Otherwise: unused
126  // If the first slot is free, it is always the head of the free list.
