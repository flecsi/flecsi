189  void make_tree() {
190  // Cstr htable
191  hmap_t hmap(hcells.span());
192 
193  // Create the tree
194  const Color size = run::context::instance().colors(),
195  rank = run::context::instance().color();
196 
197  /* Exchange high and low bound */
198  const auto hibound =
199  rank == size - 1 ? key_t::max() : data_field(2).lobound;
200  const auto lobound = rank == 0 ? key_t::min() : data_field(1).hibound;
201  assert(lobound <= e_keys(0));
202  assert(hibound >= e_keys(data_field(0).nents - 1));
203 
204  // Add the root
205  hmap.insert(key_t::root(), key_t::root());
206  assert(hmap.find(key_t::root()) != hmap.end());
207 
208  size_t current_depth = key_t::max_depth();
209  // Entity keys, last and current
210  key_t lastekey = key_t(0);
211  if(rank != 0)
212  lastekey = lobound;
213  // Node keys, last and Current
214  key_t lastnkey = key_t::root();
215  key_t nkey, loboundnode, hiboundnode;
216  // Current parent and value
217  hcell_t * parent = nullptr;
218  bool old_is_ent = false;
219 
220  bool iam0 = rank == 0;
221  bool iamlast = rank == size - 1;
222 
223  // The extra turn in the loop is to finish the missing
224  // parent of the last entity
225  for(size_t i = 0; i <= e_keys.span().size(); ++i) {
226  const key_t ekey = i < e_keys.span().size() ? e_keys(i) : hibound;
227  nkey = ekey;
228  nkey.pop(current_depth);
229  bool loopagain = false;
230  // Loop while there is a difference in the current keys
231  while(nkey != lastnkey || (iamlast && i == e_keys.span().size())) {
232  loboundnode = lobound;
233  loboundnode.pop(current_depth);
234  hiboundnode = hibound;
235  hiboundnode.pop(current_depth);
236  if(loopagain && (iam0 || lastnkey > loboundnode) &&
237  (iamlast || lastnkey < hiboundnode)) {
238  // This node is done, we can compute CoFM
239  finish_subtree(lastnkey, hmap);
240  }
241  if(iamlast && lastnkey == key_t::root())
242  break;
243  loopagain = true;
244  current_depth++;
245  nkey = ekey;
246  nkey.pop(current_depth);
247  lastnkey = lastekey;
248  lastnkey.pop(current_depth);
249  } // while
250 
251  if(iamlast && i == e_keys.span().size())
252  break;
253 
254  parent = &(hmap.find(lastnkey)->second);
255  old_is_ent = parent->is_ent();
256  // Insert the eventual missing parents in the tree
257  // Find the current parent of the two entities
258  while(1) {
259  current_depth--;
260  lastnkey = lastekey;
261  lastnkey.pop(current_depth);
262  nkey = ekey;
263  nkey.pop(current_depth);
264  if(nkey != lastnkey)
265  break;
266  // Add a children
267  int bit = nkey.last_value();
268  parent->add_child(bit);
269  parent->unset();
270  hmap.insert(nkey, nkey);
271  parent = &(hmap.find(nkey)->second);
272  } // while
273 
274  // Recover deleted entity
275  if(old_is_ent) {
276  int bit = lastnkey.last_value();
277  parent->add_child(bit);
278  parent->unset();
279  auto it = hmap.insert(lastnkey, lastnkey);
280  it->second.set_ent_idx(i - 1);
281  } // if
282 
283  if(i < e_keys.span().size()) {
284  // Insert the new entity
285  int bit = nkey.last_value();
286  parent->add_child(bit);
287  auto it = hmap.insert(nkey, nkey);
288  it->second.set_ent_idx(i);
289  } // if
290 
291  // Prepare next loop
292  lastekey = ekey;
293  lastnkey = nkey;
294  data_field(0).max_depth =
295  std::max(data_field(0).max_depth, current_depth);
296 
297  } // for
298  }
