544  virtual void map_task(const Legion::Mapping::MapperContext ctx,
545  const Legion::Task & task,
546  const Legion::Mapping::Mapper::MapTaskInput &,
547  Legion::Mapping::Mapper::MapTaskOutput & output) {
548 
549  using namespace Legion;
550  using namespace Legion::Mapping;
551  using namespace mapper;
552 
553  if(task.tag & prefer_gpu && !local_gpus.empty()) {
554  output.chosen_variant = find_gpu_variant(ctx, task.task_id);
555  output.target_procs.push_back(task.target_proc);
556  }
557  else if(task.tag & prefer_omp && !local_omps.empty()) {
558  output.chosen_variant = find_omp_variant(ctx, task.task_id);
559  output.target_procs = local_omps;
560  }
561  else {
562  output.chosen_variant = find_cpu_variant(ctx, task.task_id);
563  output.target_procs = local_cpus;
564  }
565 
566  output.chosen_instances.resize(task.regions.size());
567 
568  if(task.regions.size() > 0) {
569 
570  Legion::Memory target_mem;
571  // =
572  // DefaultMapper::default_policy_select_target_memory(
573  // ctx, task.target_proc, task.regions[0]);
574 
575  if(task.tag & prefer_gpu && !local_gpus.empty())
576  target_mem = local_framebuffer;
577  else
578  target_mem = local_sysmem;
579 
580  // creating ordering constraint (SOA )
581  std::vector<Legion::DimensionKind> ordering;
582  ordering.push_back(Legion::DimensionKind::DIM_Y);
583  ordering.push_back(Legion::DimensionKind::DIM_X);
584  ordering.push_back(Legion::DimensionKind::DIM_F); // SOA
585  Legion::OrderingConstraint ordering_constraint(
586  ordering, true /*contiguous*/);
587 
588  for(size_t indx = 0; indx < task.regions.size(); indx++) {
589 
590  // Filling out "layout_constraints" with the defaults
591  Legion::LayoutConstraintSet layout_constraints;
592  // No specialization
593  layout_constraints.add_constraint(Legion::SpecializedConstraint());
594  layout_constraints.add_constraint(ordering_constraint);
595  // Constrained for the target memory kind
596  layout_constraints.add_constraint(
597  Legion::MemoryConstraint(target_mem.kind()));
598  // Have all the field for the instance available
599  std::vector<Legion::FieldID> all_fields;
600  for(auto fid : task.regions[indx].privilege_fields) {
601  all_fields.push_back(fid);
602  } // for
603  layout_constraints.add_constraint(
604  Legion::FieldConstraint(all_fields, true));
605 
606  // creating physical instance for the reduction task
607  if(task.regions[indx].privilege == REDUCE) {
608  creade_reduction_instance(ctx, task, output, target_mem, indx);
609  }
610  else if(task.regions[indx].tag == mapper::exclusive_lr) {
611 
612  create_compacted_instance(
613  ctx, task, output, target_mem, layout_constraints, indx);
614  indx = indx + 2;
615  }
616  else {
617  create_instance(
618  ctx, task, output, target_mem, layout_constraints, indx);
619  } // end if
620  } // end for
621 
622  } // end if
623 
624  runtime->acquire_instances(ctx, output.chosen_instances);
625

628  virtual void slice_task(const Legion::Mapping::MapperContext,
629  const Legion::Task & task,
630  const Legion::Mapping::Mapper::SliceTaskInput & input,
631  Legion::Mapping::Mapper::SliceTaskOutput & output) {
632 
633  using namespace Legion;
634  using namespace mapper;
635 
636  switch(task.tag) {
637  case subrank_launch:
638  // expect a 1-D index domain
639  assert(input.domain.get_dim() == 1);
640  // send the whole domain to our local processor
641  output.slices.resize(1);
642  output.slices[0].domain = input.domain;
643  output.slices[0].proc = task.target_proc;
644  break;
645 
646  case force_rank_match:
647  case compacted_storage: {
648  // expect a 1-D index domain - each point goes to the corresponding node
649  assert(input.domain.get_dim() == 1);
650  LegionRuntime::Arrays::Rect<1> r = input.domain.get_rect<1>();
651 
652  // go through all the CPU processors and find a representative for each
653  // node (i.e. address space)
654  std::map<int, Legion::Processor> targets;
655 
656  Legion::Machine::ProcessorQuery pq =
657  Legion::Machine::ProcessorQuery(machine).only_kind(
658  Legion::Processor::LOC_PROC);
659  for(Legion::Machine::ProcessorQuery::iterator it = pq.begin();
660  it != pq.end();
661  ++it) {
662  Legion::Processor p = *it;
663  int a = p.address_space();
664  if(targets.count(a) == 0)
665  targets[a] = p;
666  }
667 
668  output.slices.resize(1);
669  for(int a = r.lo[0]; a <= r.hi[0]; a++) {
670  assert(targets.count(a) > 0);
671  output.slices[0].domain = // Legion::Domain::from_rect<1>(
672  Legion::Rect<1>(a, a);
673  output.slices[0].proc = targets[a];
674  }
675  break;
676  }
677 
678  default:
679  // We've already been control replicated, so just divide our points
680  // over the local processors, depending on which kind we prefer
681  if(task.tag == prefer_gpu && !local_gpus.empty()) {
682  unsigned local_gpu_index = 0;
683  for(Domain::DomainPointIterator itr(input.domain); itr; itr++) {
684  TaskSlice slice;
685  slice.domain = Domain(itr.p, itr.p);
686  slice.proc = local_gpus[local_gpu_index++];
687  if(local_gpu_index == local_gpus.size())
688  local_gpu_index = 0;
689  slice.recurse = false;
690  slice.stealable = false;
691  output.slices.push_back(slice);
692  }
693  }
694  else if(task.tag == prefer_omp && !local_omps.empty()) {
695  unsigned local_omp_index = 0;
