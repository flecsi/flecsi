61 // Construct a tuple of converted arguments (or references to existing
62 // arguments where possible). Note that is_constructible_v<const
63 // float&,const double&> is true, so we have to check
64 // is_constructible_v<float&,double&> instead.
65 template<bool M, class... PP, class... AA>
66 std::conditional_t<M,
67  std::tuple<PP...>,
68  std::tuple<std::conditional_t<
69  std::is_constructible_v<nonconst_ref_t<PP> &, nonconst_ref_t<AA>>,
70  const PP &,
71  std::decay_t<PP>>...>>
72 make_parameters(std::tuple<PP...> * /* to deduce PP */, AA &&... aa) {
73  static_assert(
74  M || (std::is_const_v<std::remove_reference_t<const PP>> && ...),
75  "only MPI tasks can accept non-const references");
76  return {exec::replace_argument<PP>(std::forward<AA>(aa))...};
77 }

86 template<auto & F, class Reduction, TaskAttributes Attributes, typename... Args>
87 auto
88 reduce_internal(Args &&... args) {
89  using namespace Legion;
90  using traits_t = util::function_traits<decltype(F)>;
91  using return_t = typename traits_t::return_type;
92  using param_tuple = typename traits_t::arguments_type;
93 
94  // This will guard the entire method
95  log::devel_guard guard(execution_tag);
96 
97  // Get the FleCSI runtime context
98  auto & flecsi_context = run::context::instance();
99 
100  // Get the processor type.
101  constexpr auto processor_type = mask_to_processor_type(Attributes);
102 
103  // Get the Legion runtime and context from the current task.
104  auto legion_runtime = Legion::Runtime::get_runtime();
105  auto legion_context = Legion::Runtime::get_context();
106 
107  constexpr bool mpi_task = processor_type == task_processor_type_t::mpi;
108  static_assert(processor_type == task_processor_type_t::toc ||
109  processor_type == task_processor_type_t::loc || mpi_task,
110  "Unknown launch type");
111  const auto domain_size =
112  launch_size<Attributes, param_tuple>(std::forward<Args>(args)...);
113 
114  auto params =
115  detail::make_parameters<mpi_task, param_tuple>(std::forward<Args>(args)...);
116  prolog pro(params, args...);
117 
118  std::vector<std::byte> buf;
119  if constexpr(mpi_task) {
120  // MPI tasks must be invoked collectively from one task on each rank.
121  // We therefore can transmit merely a pointer to a tuple of the arguments.
122  // The TaskArgument must be identical on every shard, so use the context.
123  flecsi_context.mpi_params = &params;
124  }
125  else {
126  buf = std::apply(
127  [](const auto &... pp) { return util::serial_put_tuple(pp...); }, params);
128  }
129 
130  using wrap = leg::task_wrapper<F, processor_type>;
131  // Replace the MPI "processor type" with an actual flag:
132  const auto task = leg::task_id<wrap::execute,
133  Attributes & ~mpi | as_mask(wrap::LegionProcessor)>;
134 
135  const auto add = [&](auto & l) {
136  for(auto & req : pro.region_requirements())
137  l.add_region_requirement(req);
138  l.futures = std::move(pro).futures();
139  if(processor_type == task_processor_type_t::toc)
140  l.tag = run::mapper::prefer_gpu;
141  };
142 
143  if constexpr(std::is_same_v<decltype(domain_size), const std::monostate>) {
144  {
145  log::devel_guard guard(execution_tag);
146  flog_devel(info) << "Executing single task" << std::endl;
147  }
148 
149  TaskLauncher launcher(task, TaskArgument(buf.data(), buf.size()));
150  add(launcher);
151 
152  return future<return_t>{
153  legion_runtime->execute_task(legion_context, launcher)};
154  }
155  else {
156  {
157  log::devel_guard guard(execution_tag);
158  flog_devel(info) << "Executing index task" << std::endl;
159  }
160 
161  LegionRuntime::Arrays::Rect<1> launch_bounds(
162  LegionRuntime::Arrays::Point<1>(0),
163  LegionRuntime::Arrays::Point<1>(domain_size - 1));
164  Domain launch_domain = Domain::from_rect<1>(launch_bounds);
165 
166  Legion::ArgumentMap arg_map;
167  Legion::IndexLauncher launcher(
168  task, launch_domain, TaskArgument(buf.data(), buf.size()), arg_map);
169  add(launcher);
170  launcher.point_futures.assign(
171  pro.future_maps().begin(), pro.future_maps().end());
172 
173  if(mpi_task)
174  launcher.tag = run::mapper::force_rank_match;
175 
176  if constexpr(!std::is_void_v<Reduction>) {
177  flog_devel(info) << "executing reduction logic for "
178  << util::type<Reduction>() << std::endl;
179 
180  auto ret = future<return_t, launch_type_t::single>{
181  legion_runtime->execute_index_space(
182  legion_context, launcher, fold::wrap<Reduction, return_t>::id)};
183  if(mpi_task)
184  ret.wait();
185  return ret;
186  }
187  else {
188  auto ret = future<return_t, launch_type_t::index>{
189  legion_runtime->execute_index_space(legion_context, launcher)};
190  if(mpi_task)
191  ret.wait();
192 
193  return ret;
194  } // if reduction
195 
196  } // if constexpr

198 } // reduce_internal
