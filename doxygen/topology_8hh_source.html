71  // Return whether a copy is needed.
72  template<Privileges P>
73  bool ghost(field_id_t i) {
74  constexpr auto n = privilege_count(P);
75  static_assert(n > 1, "need shared/ghost privileges");
76  constexpr auto g = get_privilege(n - 1, P);
77  constexpr bool gr = privilege_read(g),
78  sw = privilege_write(get_privilege(n - 2, P));
79  // The logic here is constructed to allow a single read/write set access:
80  // writing to ghosts, or reading from them without also writing to shared,
81  // clears the dirty bit, and otherwise writing to shared sets it.
82  // Otherwise, it retains its value (and we don't copy).
83  return (privilege_write(g) || !sw && gr ? dirty.erase(i)
84  : sw && !dirty.insert(i).second) &&
85  gr;
86  }
