31 struct region_impl {
32  // The constructor is collectively called on all ranks with the same s,
33  // and fs. s.first is number of rows while s.second is number of columns.
34  // MPI may assume "number of rows" == number of ranks. The number of columns
35  // is a placeholder for the number of data points to be stored in a
36  // particular row (aka rank), it could be exact or in the case when we don't
37  // know the exact number yet, a large number (flecsi::data::logical_size) is
38  // used.
39  //
40  // Generic frontend code supplies `s` and `fs` (for information about fields),
41  // requesting memory to be (notionally) reserved from backend. Here we only
42  // create the underlying std::vector<> without actually allocating any memory.
43  // The client code (mostly exec::task_prologue) will call .get_storage() with
44  // a field id and the number of elements on this rank, as determined by
45  // partitioning of the field. We will then call .resize() on the
46  // std::vector<>.
47  region_impl(size2 s, const fields & fs) : s(std::move(s)), fs(fs) {
48  for(auto f : fs) {
49  storages.emplace(f->fid, 0);
50  }
51  }
52 
53  size2 size() const {
54  return s;
55  }
56 
57  // The span is safe because it is used only within a user task while the
58  // vectors are resized or destroyed only outside user tasks (though perhaps
59  // during execute).
60  template<class T>
61  util::span<T> get_storage(field_id_t fid) {
62  return get_storage<T>(fid, s.second);
63  }
64 
65  template<class T>
66  util::span<T> get_storage(field_id_t fid, std::size_t nelems) {
67  auto & v = storages.at(fid);
68  std::size_t nbytes = nelems * sizeof(T);
69  if(nbytes > v.size())
70  v.resize(nbytes);
71  return {reinterpret_cast<T *>(v.data()), nelems};
72  }
73 
74  auto get_field_info(field_id_t fid) const {
75  for(auto f : fs) {
76  if(f->fid == fid)
77  return f;
78  }
79  throw std::runtime_error("can not find field");
80  }
81 
82 private:
83  size2 s; // (nranks, nelems)
84  fields fs; // fs[].fid is only unique within a region, i.e. r0.fs[].fid is
85  // unrelated to r1.fs[].fid even if they have the same value.
86 
87  std::unordered_map<field_id_t, std::vector<std::byte>> storages;
88 };

172 struct intervals {
173  using Value = subrow; // [begin, end)
174  static Value make(subrow r, std::size_t = 0) {
175  return r;
176  }
177 
178  intervals(region_base & r,
179  const partition & p,
180  field_id_t fid, // The field id for the metadata in the region in p.
181  completeness = incomplete)
182  : r(&*r) {
183  // Called by upper layer, supplied with a region and a partition. There are
184  // two regions involved. The region `r` has the storage for real field data
185  // (e.g. density, pressure etc.) as the destination of the ghost copy. It
186  // also contains the pairs of (rank, index) of shared entities on remote
187  // peers. The region and associated storage in the partition `p` contains
188  // metadata on which entities are local ghosts (destination of copy) on the
189  // current rank. The metadata is in the form of [beginning index, ending
190  // index), type aliased as Value, into the index space of the entity (e.g.
191  // vertex, edge, cell). We thus need to use the p.get_storage() to get the
192  // metadata, not the region.get_storage() which gives the real data. This
193  // works the same way as how one partition stores the number of elements in
194  // a particular 'shard' on a rank for another partition. In addition, we
195  // also need to copy Values from the partition and save them locally. User
196  // code might change it after this constructor returns. We can not use a
197  // copy assignment directly here since metadata is an util::span while
198  // ghost_ranges is a std::vector<>.
199  ghost_ranges = to_vector(p.get_storage<Value>(fid));
200  // Get The largest value of `end index` in ghost_ranges (i.e. the upper
201  // bound). This tells how much memory needs to be allocated for ghost
202  // entities.
203  if(auto iter = std::max_element(ghost_ranges.begin(),
204  ghost_ranges.end(),
205  [](Value x, Value y) { return x.second < y.second; });
206  iter != ghost_ranges.end()) {
207  max_end = iter->second;
208  }
209  }
210 
211 private:
212  // This member function is only called by copy_engine.
213  friend struct copy_engine;
214 
215  template<typename T>
216  auto get_storage(field_id_t fid) const {
217  return r->get_storage<T>(fid, max_end);
218  }
219 
220  // We use a reference to region instead of pointer since it is not nullable
221  // nor needs to be redirected after initialization.
222  mpi::region_impl * r;
223 
224  // Locally cached metadata on ranges of ghost index.
225  std::vector<Value> ghost_ranges;
226  std::size_t max_end = 0;
227 };

250 struct copy_engine {
251  // One copy engine for each entity type i.e. vertex, cell, edge.
252  copy_engine(const points & points,
253  const intervals & intervals,
254  field_id_t meta_fid /* for remote shared entities */)
255  : source(points), destination(intervals), meta_fid(meta_fid) {
256  // There is no information about the indices of local shared entities,
257  // ranks and indices of the destination of copy i.e. (local source
258  // index, {(remote dest rank, remote dest index)}). We need to do a shuffle
259  // operation to reconstruct this info from {(local ghost index, remote
260  // source rank, remote source index)}.
261  auto remote_sources =
262  destination.get_storage<const points::Value>(meta_fid);
263  // Essentially a GroupByKey of remote_sources, keys are the remote source
264  // ranks and values are vectors of remote source indices.
265  SendPoints grouped_shared_entities;
266  for(const auto & [begin, end] : destination.ghost_ranges) {
267  for(auto ghost_idx = begin; ghost_idx < end; ++ghost_idx) {
268  const auto & shared = remote_sources[ghost_idx];
269  grouped_shared_entities[shared.first].emplace_back(shared.second);
270  }
271  }
272 
273  // Create the inverse mapping of
274  // group_shared_entities. This creates a map from remote destination rank to
275  // a vector of *local* source indices. This information is later used by
276  // MPI_Send().
277  {
278  std::size_t r = 0;
279  for(auto &v : util::mpi::all_to_allv([&](int r, int) -> auto & {
280  static const std::vector<std::size_t> empty;
281  const auto i = grouped_shared_entities.find(r);
282  return i == grouped_shared_entities.end() ? empty : i->second;
283  })) {
284  if(!v.empty())
285  remote_ghost_entities.try_emplace(r, std::move(v));
286  ++r;
287  }
288  }
289 
290  // We need to figure out the max local source index in order to give correct
291  // nelems when calling region::get_storage().
292  for(const auto & [rank, indices] : remote_ghost_entities) {
293  max_local_source_idx = std::max(max_local_source_idx,
294  *std::max_element(indices.begin(), indices.end()));
295  }
296  max_local_source_idx += 1;
297 
298  // We need to reserve enough memory for MPI requests used in operator().
299  // This only need to be calculated once.
300  auto nrecvs = std::transform_reduce(destination.ghost_ranges.begin(),
301  destination.ghost_ranges.end(),
302  0,
303  std::plus<>(), // this uses C++14 std::plus<void> where T is deduced.
304  [](const auto & p) { return p.second - p.first; });
305  auto nsends = std::transform_reduce(remote_ghost_entities.begin(),
306  remote_ghost_entities.end(),
307  0,
308  std::plus<>(), // this uses C++14 std::plus<void> where T is deduced.
309  [](const auto & p) { return p.second.size(); });
310 
311  nreqs = nrecvs + nsends;
312  }
313 
314  // called with each field (and field_id_t) on the entity, for example, one
315  // for pressure, temperature, density etc.
316  void operator()(field_id_t data_fid) const {
317  using util::mpi::test;
318 
319  auto source_storage =
320  source.r->get_storage<std::byte>(data_fid, max_local_source_idx);
321  auto destination_storage = destination.get_storage<std::byte>(data_fid);
322 
323  // FIXME: should we assert(source_type_size == dest_type_size)?
324  auto type_size = source.r->get_field_info(data_fid)->type_size;
325 
326  std::vector<MPI_Request> requests;
327  requests.reserve(nreqs);
328 
329  auto remote_sources =
330  destination.get_storage<const points::Value>(meta_fid);
331 
332  for(const auto & [begin, end] : destination.ghost_ranges) {
333  for(auto ghost_idx = begin; ghost_idx < end; ++ghost_idx) {
334  auto source_rank = remote_sources[ghost_idx].first;
335  requests.resize(requests.size() + 1);
336  test(MPI_Irecv(destination_storage.data() + ghost_idx * type_size,
337  type_size,
338  MPI_BYTE,
339  source_rank,
340  0,
341  MPI_COMM_WORLD,
342  &requests.back()));
343  }
344  }
345 
346  for(const auto & [dest_rank, local_indices] : remote_ghost_entities) {
347  for(auto shared_idx : local_indices) {
348  requests.resize(requests.size() + 1);
349  test(MPI_Isend(source_storage.data() + shared_idx * type_size,
350  type_size,
351  MPI_BYTE,
352  int(dest_rank),
353  0,
354  MPI_COMM_WORLD,
355  &requests.back()));
356  }
357  }
358 
359  std::vector<MPI_Status> status(requests.size());
360  test(MPI_Waitall(requests.size(), requests.data(), status.data()));
361  }
362 
363 private:
364  // (rank, { indices })
365  using SendPoints = std::map<std::size_t, std::vector<std::size_t>>;
366 
367  const points & source;
368  const intervals & destination;
369  field_id_t meta_fid;
370  SendPoints remote_ghost_entities;
371  std::size_t max_local_source_idx = 0;
372  std::size_t nreqs = 0;
373 };
