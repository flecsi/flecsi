177 template<typename RETURN, task<RETURN> * TASK, TaskAttributes A>
178 void
179 detail::register_task() {
180  constexpr auto processor_type = mask_to_processor_type(A);
181  static_assert(processor_type != task_processor_type_t::mpi,
182  "Legion tasks cannot use MPI");
183 
184  const std::string name = util::symbol<*TASK>();
185  {
186  log::devel_guard guard(task_wrapper_tag);
187  flog_devel(info) << "registering pure Legion task " << name << std::endl;
188  }
189 
190  Legion::TaskVariantRegistrar registrar(task_id<*TASK, A>, name.c_str());
191  Legion::Processor::Kind kind = processor_type == task_processor_type_t::toc
192  ? Legion::Processor::TOC_PROC
193  : Legion::Processor::LOC_PROC;
194  registrar.add_constraint(Legion::ProcessorConstraint(kind));
195 
196  registrar.set_leaf(A & leaf || ~A & inner);
197  registrar.set_inner(A & inner);
198  registrar.set_idempotent(A & idempotent);
199 
200  /*
201  This section of conditionals is necessary because there is still
202  a distinction between void and non-void task registration with
203  Legion.
204  */
205 
206  if constexpr(std::is_same_v<RETURN, void>) {
207  Legion::Runtime::preregister_task_variant<TASK>(registrar, name.c_str());
208  }
209  else {
210  Legion::Runtime::preregister_task_variant<RETURN, TASK>(
211  registrar, name.c_str());
212  } // if
213 }

248  static RETURN execute(const Legion::Task * task,

250  Legion::Context context,

252  {

254  flog_devel(info) << "In execute_user_task" << std::endl;

257  // Unpack task arguments
