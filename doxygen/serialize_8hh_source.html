67 template<class F>
68 auto serial_buffer(F && f) { // f should accept a P for serial_put
69  std::size_t sz = 0;
70  f(sz);
71  std::vector<std::byte> ret(sz);
72  auto *const p0 = ret.data(), *p = p0;
73  f(p);
74  flog_assert(p == p0 + sz, "Wrong serialization size");

121  serial_put(p, c.size());

123  serial_put(p, t);

143 template<class T>
144 struct serial<T, std::enable_if_t<bit_copyable_v<T>>> {
145  static_assert(!std::is_pointer_v<T>, "Cannot serialize pointers");
146  template<class P>
147  static void put(P & p, const T & t) {
148  mempcpy(p, &t, sizeof t);
149  }
150  static T get(const std::byte *& p) {
151  T ret;
152  // Suppress -Wclass-memaccess: the default constructor needn't be trivial!
153  std::memcpy(static_cast<void *>(&ret), p, sizeof ret);
154  p += sizeof ret;
155  return ret;
156  }
