59  static Legion::PrivilegeMode privilege_mode(Privileges mode) {
60  // Reduce the read and write permissions for each privilege separately:
61  bool r = false, w = false;
62  for(auto i = privilege_count(mode); i-- && !(r && w);) {
63  const auto p = get_privilege(i, mode);
64  r = r || privilege_read(p);
65  w = w || privilege_write(p);
66  }
67  return r ? w ? READ_WRITE : READ_ONLY
68  : w ? privilege_discard(mode) ? WRITE_DISCARD : WRITE_ONLY
69  : NO_ACCESS;
70  } // privilege_mode
