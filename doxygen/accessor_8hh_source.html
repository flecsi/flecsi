421  if(i < b) {

423  // used elements assigned from t:

426  fillc = std::min(fillx, o->del), // slack spaces constructed from t

634  void commit() const {
635  // To move each element before overwriting it, we propagate moves outward
636  // from each place where the movement switches from rightward to leftward.
637  const auto all = acc.get_base().span();
638  const size_type n = size();
639  // Read and write cursors. It would be possible, if ugly, to run cursors
640  // backwards for the rightward-moving portions and do without the stack.
641  size_type is = 0, id = 0;
642  base_size js = 0, jd = 0;
643  raw_row rs = raw_get(is), rd = raw_get(id);
644  struct work {
645  void operator()() const {
646  if(assign)
647  *w = std::move(*r);
648  else
649  new(w) T(std::move(*r));
650  }
651  T *r, *w;
652  bool assign;
653  };
654  std::stack<work> stk;
655  const auto back = [&stk] {
656  for(; !stk.empty(); stk.pop())
657  stk.top()();
658  };
659  for(;; ++js, ++jd) {
660  while(js == rs.size()) { // js indexes the split row
661  if(++is == n)
662  break;
663  rs = raw_get(is);
664  js = 0;
665  }
666  if(is == n)
667  break;
668  while(jd == rd.s.size()) { // jd indexes the span (including slack space)
669  if(++id == n)
670  break; // we can write off the end of the last
671  else
672  rd = raw_get(id);
673  jd = 0;
674  }
675  const auto r = &rs[js], w = rd.s.data() + jd;
676  flog_assert(w != all.end(),
677  "ragged entries for last " << n - is << " rows overrun allocation for "
678  << all.size() << " entries");
679  if(r != w)
680  stk.push({r, w, jd < rd.brk()});
681  // Perform this move, and any already queued, if it's not to the right.
682  // Offset real elements by one position to come after insertions.
683  if(rs.s.data() + std::min(js + 1, rs.brk()) > w)
684  back();
685  }
686  back();
687  if(jd < rd.brk())
688  rd.destroy(jd);
