158 // A "strong typedef" for T that supports overload resolution, template
159 // argument deduction, and limited arithmetic.
160 // The first parameter differentiates topologies/index spaces.
161 template<auto, class T = util::id>
162 struct id {
163  using difference_type = std::make_signed_t<T>;
164 
165  id() = default; // allow trivial default initialization
166  FLECSI_INLINE_TARGET
167  explicit id(T t) : t(t) {}
168 
169  FLECSI_INLINE_TARGET
170  T operator+() const {
171  return t;
172  }
173 
174  FLECSI_INLINE_TARGET
175  operator T() const {
176  return t;
177  }
178 
179  // Prevent assigning to transform_view results:
180  id & operator=(const id &) & = default;
181 
182  FLECSI_INLINE_TARGET
183  id & operator++() & {
184  ++t;
185  return *this;
186  }
187  FLECSI_INLINE_TARGET
188  id operator++(int) & {
189  id ret = *this;
190  ++*this;
191  return ret;
192  }
193  FLECSI_INLINE_TARGET
194  id & operator--() & {
195  --t;
196  return *this;
197  }
198  FLECSI_INLINE_TARGET
199  id operator--(int) & {
200  id ret = *this;
201  --*this;
202  return ret;
203  }
204 
205  FLECSI_INLINE_TARGET
206  id & operator+=(difference_type d) & {
207  t += d;
208  return *this;
209  }
210  FLECSI_INLINE_TARGET
211  id operator+(difference_type d) const {
212  return d + *this;
213  }
214  FLECSI_INLINE_TARGET
215  void operator+(id) const = delete;
216  friend id operator+(difference_type d, id i) {
217  return i += d;
218  }
219  FLECSI_INLINE_TARGET
220  id & operator-=(difference_type d) & {
221  t -= d;
222  return *this;
223  }
224  FLECSI_INLINE_TARGET
225  id operator-(difference_type d) const {
226  return id(difference_type(*this) - d);
227  }
228  FLECSI_INLINE_TARGET
229  difference_type operator-(id i) const { // also avoids ambiguity
230  return difference_type(t) - difference_type(i.t);
231  }
232 
233 private:
234  T t;
235 };
