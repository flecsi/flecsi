108 inline std::string
109 timestamp(bool underscores = false) {
110  char stamp[14];
111  time_t t = time(0);
112  std::string format = underscores ? "%m%d_%H%M%S" : "%m%d %H:%M:%S";
113  strftime(stamp, sizeof(stamp), format.c_str(), localtime(&t));
114  return std::string(stamp);
115 } // timestamp
