42 template<typename MD>
43 inline auto
44 make_dcrs(MD const & md,
45  Dimension through_dimension,
46  MPI_Comm comm = MPI_COMM_WORLD) {
47  auto [rank, size] = util::mpi::info(comm);
48 
49  std::size_t nc = md.num_entities(MD::dimension());
50  std::size_t nv = md.num_entities(0);
51 
52  util::color_map cm(size, size, nc);
53 
54  /*
55  Get the initial cells for this rank. The cells will be read by
56  the root process and sent to each initially "owning" rank using
57  a naive distribution.
58  */
59 
60  auto c2v =
61  util::mpi::one_to_allv<pack_cells<MD>>({md, cm.distribution()}, comm);
62 
63  /*
64  Create a map of vertex-to-cell connectivity information from
65  the initial cell distribution.
66  */
67 
68  // Populate local vertex connectivity information
69  std::size_t offset{cm.distribution()[rank]};
70  std::size_t indices{cm.indices(rank, 0)};
71  std::map<std::size_t, std::vector<std::size_t>> v2c;
72 
73  std::size_t i{0};
74  for(auto c : c2v) {
75  for(auto v : c) {
76  v2c[v].emplace_back(offset + i);
77  } // for
78  ++i;
79  } // for
80 
81  // Request all referencers of our connected vertices
82  util::color_map vm(size, size, nv);
83  auto referencers = util::mpi::all_to_allv<vertex_referencers>(
84  {v2c, vm.distribution(), rank}, comm);
85 
86  /*
87  Update our local connectivity information. We now have all
88  vertex-to-cell connectivity informaiton for the naive distribution
89  of cells that we own.
90  */
91 
92  i = 0;
93  for(auto & r : referencers) {
94  for(auto v : r) {
95  for(auto c : v.second) {
96  v2c[v.first].emplace_back(c);
97  } // for
98  } // for
99  ++i;
100  } // for
101 
102  // Remove duplicate referencers
103  util::unique_each(v2c);
104 
105  std::vector<std::vector<std::size_t>> referencer_inverse(size);
106 
107  for(auto const & v : v2c) {
108  for(auto c : v.second) {
109  auto r = util::distribution_offset(cm.distribution(), c);
110  if(r != rank) {
111  referencer_inverse[r].emplace_back(v.first);
112  } // if
113  } // for
114  } // for
115 
116  // Remove duplicate inverses
117  util::unique_each(referencer_inverse);
118 
119  // Request vertex-to-cell connectivity for the cells that are
120  // on other ranks in the naive cell distribution.
121  auto connectivity = util::mpi::all_to_allv<cell_connectivity>(
122  {referencer_inverse, v2c, vm.distribution(), rank}, comm);
123 
124  for(auto & r : connectivity) {
125  for(auto & v : r) {
126  for(auto c : v.second) {
127  v2c[v.first].emplace_back(c);
128  } // for
129  } // for
130  } // for
131 
132  // Remove duplicate referencers
133  util::unique_each(v2c);
134 
135  std::map<std::size_t, std::vector<std::size_t>> c2c;
136  std::size_t c{offset};
137  for(auto & cd /* cell definition */ : c2v) {
138  std::map<std::size_t, std::size_t> through;
139 
140  for(auto v : cd) {
141  auto it = v2c.find(v);
142  if(it != v2c.end()) {
143  for(auto rc : v2c.at(v)) {
144  if(rc != c)
145  ++through[rc];
146  } // for
147  } // if
148  } // for
149 
150  for(auto tc : through) {
151  if(tc.second > through_dimension) {
152  c2c[c].emplace_back(tc.first);
153  c2c[tc.first].emplace_back(c);
154  } // if
155  } // for
156 
157  ++c;
158  } // for
159 
160  // Remove duplicate connections
161  util::unique_each(c2c);
162 
163  util::dcrs dcrs;
164  dcrs.distribution = cm.distribution();
165 
166  dcrs.offsets.emplace_back(0);
167  for(std::size_t c{0}; c < indices; ++c) {
168  for(auto cr : c2c[offset + c]) {
169  dcrs.indices.emplace_back(cr);
170  } // for
171 
172  dcrs.offsets.emplace_back(dcrs.offsets[c] + c2c[offset + c].size());
173  } // for
174 
175  return std::make_tuple(dcrs, std::make_pair(nc, nv), c2v, v2c, c2c);
176 } // make_dcrs

201 inline auto
202 migrate(util::dcrs const & naive,
203  Color colors,
204  std::vector<std::size_t> const & index_colors,
205  std::vector<std::vector<std::size_t>> & c2v,
206  std::map<std::size_t, std::vector<std::size_t>> & v2c,
207  std::map<std::size_t, std::vector<std::size_t>> & c2c,
208  MPI_Comm comm = MPI_COMM_WORLD) {
209  auto [rank, size] = util::mpi::info(comm);
210 
211  auto migrated = util::mpi::all_to_allv<migrate_cells>(
212  {naive, colors, index_colors, c2v, v2c, c2c, rank}, comm);
213 
214  std::map<Color, std::vector<std::size_t>> primaries;
215  std::vector<std::size_t> p2m; /* process to mesh map */
216  std::map<std::size_t, std::size_t> m2p;
217 
218  for(auto const & r : migrated) {
219  auto const & cell_pack = std::get<0>(r);
220  for(auto const & c : cell_pack) { /* std::vector over cells */
221  auto const & info = std::get<0>(c); /* std::array<color, mesh id> */
222  c2v.emplace_back(std::get<1>(c)); /* cell definition (vertex mesh ids) */
223  m2p[std::get<1>(info)] = c2v.size() - 1; /* offset map */
224  p2m.emplace_back(std::get<1>(info)); /* cell mesh id */
225  primaries[std::get<0>(info)].emplace_back(std::get<1>(info));
226  } // for
227 
228  // vertex-to-cell connectivity
229  auto v2c_pack = std::get<1>(r);
230  for(auto const & v : v2c_pack) {
231  v2c.try_emplace(v.first, v.second);
232  } // for
233 
234  // cell-to-cell connectivity
235  auto c2c_pack = std::get<2>(r);
236  for(auto const & c : c2c_pack) {
237  c2c.try_emplace(c.first, c.second);
238  } // for
239  } // for
240 
241  return std::make_tuple(primaries, p2m, m2p);
242 } // migrate

246 color(MD const & md,
247  coloring_definition const & cd,
248  std::vector<std::size_t> const & raw,
249  std::map<Color, std::vector<std::size_t>> const & primaries,
250  std::vector<std::vector<std::size_t>> & e2v,
251  std::map<std::size_t, std::vector<std::size_t>> & v2e,
252  std::map<std::size_t, std::vector<std::size_t>> & e2e,
253  std::map<std::size_t, std::size_t> & m2p,
254  std::vector<std::size_t> & p2m,
255  MPI_Comm comm = MPI_COMM_WORLD) {
256  auto [rank, size] = util::mpi::info(comm);
257 
258  std::size_t ne = md.num_entities(cd.dim);
259 
260  /*
261  Save color info for all of our initial local entities. The variable
262  'e2co' will get updated as we build out our dependencies.
263  */
264 
265  std::unordered_map<std::size_t, Color> e2co;
266  std::map<std::size_t, std::vector<std::size_t>> wkset;
267  for(auto const & p : primaries) {
268  wkset[p.first].reserve(p.second.size());
269  for(auto e : p.second) {
270  e2co.try_emplace(e, p.first);
271  wkset[p.first].emplace_back(e);
272  } // for
273  } // for
274 
275  std::unordered_map<std::size_t, std::set<Color>> dependents, dependencies;
276  std::unordered_map<Color, std::set<std::size_t>> shared, ghosts;
277 
278  const std::size_t depth = cd.depth;
279  for(std::size_t d{0}; d < depth + 1; ++d) {
280  std::vector<std::size_t> layer;
281 
282  /*
283  Create request layer, and add local information.
284  */
285 
286  for(auto const & p : primaries) {
287  for(auto e : wkset.at(p.first)) {
288  for(auto v : e2v[m2p.at(e)]) {
289  for(auto en : v2e.at(v)) {
290  if(e2e.find(en) == e2e.end()) {
291  // If we don't have the entity, we need to request it.
292  layer.emplace_back(en);
293 
294  // Collect the dependent color for request.
295  dependencies[en].insert(p.first);
296 
297  // If we're within the requested depth, this is also
298  // a ghost entity.
299  if(d < depth) {
300  ghosts[p.first].insert(en);
301  }
302  }
303  else if(d < depth && e2co.at(en) != p.first) {
304  // This entity is on the local process, but not
305  // owned by the current color.
306 
307  // Add this entity to the shared of the owning color.
308  shared[e2co.at(en)].insert(en);
309 
310  // Add the current color as a dependent.
311  dependents[en].insert(p.first);
312 
313  // This entity is a ghost for the current color.
314  ghosts[p.first].insert(en);
315  } // if
316  } // for
317  } // for
318  } // for
319 
320  wkset.at(p.first).clear();
321  } // for
322 
323  util::force_unique(layer);
324 
325  /*
326  Request entity owners from naive-owners.
327  */
328 
329  std::vector<std::vector<std::size_t>> requests(size);
330  util::color_map nm(size, size, ne);
331  for(auto e : layer) {
332  requests[nm.process(nm.index_color(e))].emplace_back(e);
333  } // for

336  size);

338  auto requested = util::mpi::all_to_allv(
339  [&requests](int r, int) -> auto & { return requests[r]; }, comm);
340

342  Fulfill naive-owner requests with migrated owners.
343  */
344

346  {

348  util::color_map cm(size, cd.colors, ne);
349  for(auto rv : requested) {
350  for(auto e : rv) {
351  const std::size_t start = nm.index_offset(rank, 0);
352  fulfills[r].emplace_back(cm.process(raw[e - start]));
353  } // for
354  ++r;
355  } // for
356  } // scope

358  auto fulfilled = util::mpi::all_to_allv(
359  [&fulfills](int r, int) -> auto & { return fulfills[r]; }, comm);
360 
361  /*
362  Request entity information from migrated owners.
363  */
364 
365  std::vector<std::size_t> offs(size, 0ul);
366  for(auto e : layer) {
367  auto p = nm.process(nm.index_color(e));
368  reqs[fulfilled[p][offs[p]++]].emplace_back(
369  std::make_pair(e, dependencies.at(e)));
370  } // for
371  } // scope
372 
373  auto requested = util::mpi::all_to_allv(
374  [&reqs](int r, int) -> auto & { return reqs[r]; }, comm);
375 
376  /*
377  Keep track of dependent colors for requested entities.
378  */
379 
380  requests.clear();
381  requests.resize(size);
382  Color r = 0;
383  for(auto rv : requested) {
384  for(auto e : rv) {
385  requests[r].emplace_back(e.first);
386  if(d < depth) {
387  dependents[e.first].insert(e.second.begin(), e.second.end());
388  shared[e2co.at(e.first)].insert(e.first);
389  }
390  } // for
391  ++r;
392  } // for

395  {requests, e2co, e2v, v2e, e2e, m2p}, comm);
396

399  */
400 
401  for(auto const & r : fulfilled) {
402  auto const & entity_pack = std::get<0>(r);
403  for(auto const & e : entity_pack) {
404  auto const & info = std::get<0>(e);
405  e2v.emplace_back(std::get<1>(e));
406  m2p[std::get<1>(info)] = e2v.size() - 1;
407  p2m.emplace_back(std::get<1>(info));
408  e2co.try_emplace(std::get<1>(info), std::get<0>(info));
409 
410  for(auto co : dependencies.at(std::get<1>(info))) {
411  wkset.at(co).emplace_back(std::get<1>(info));
412  } // for
413  } // for
414 
415  // vertex-to-cell connectivity
416  auto v2e_pack = std::get<1>(r);
417  for(auto const & v : v2e_pack) {
418  v2e.try_emplace(v.first, v.second);
419  } // for
420 
421  // cell-to-cell connectivity
422  auto e2e_pack = std::get<2>(r);
423  for(auto const & e : e2e_pack) {
424  e2e.try_emplace(e.first, e.second);
425  } // for
426  } // for
427  } // for
