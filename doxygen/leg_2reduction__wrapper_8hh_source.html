41 struct atomic_base {
42  // Expecting that concurrent atomic operations are more likely on a single
43  // type, we share one large lock array:
44  static constexpr std::uintptr_t locks = 255;
45  static inline std::mutex lock[locks];
46 
47  static auto lower(std::memory_order o) {
48  switch(o) {
49  case std::memory_order_relaxed:
50  return __ATOMIC_RELAXED;
51  case std::memory_order_consume:
52  return __ATOMIC_CONSUME;
53  case std::memory_order_acquire:
54  return __ATOMIC_ACQUIRE;
55  case std::memory_order_release:
56  return __ATOMIC_RELEASE;
57  case std::memory_order_acq_rel:
58  return __ATOMIC_ACQ_REL;
59  default:
60  return __ATOMIC_SEQ_CST;
61  }
62  }

65 template<class T, class = void>
66 struct atomic_ref : private atomic_base {
67  explicit atomic_ref(T & t) : p(&t) {}
68  bool compare_exchange_strong(T & expected,
69  T desired,
70  std::memory_order = {}) const noexcept {
71  const std::unique_lock guard(
72  lock[reinterpret_cast<std::uintptr_t>(p) / alignof(T) % locks]);
73  const bool fail = std::memcmp(p, &expected, sizeof(T));
74  std::memcpy(fail ? &expected : p, fail ? p : &desired, sizeof(T));
75  return !fail;
76  }
77 
78 private:
79  T * p;
80 };

113 // Adapts our interface to Legion's.
114 template<class R, class T, class = void>
115 struct wrap {
116  typedef T LHS, RHS;
117 
118  template<bool E>
119  static void apply(LHS & a, RHS b) {
120  if constexpr(E)
121  a = R::combine(a, b);
122  else {
123  LHS rd{};
124  detail::atomic_ref<LHS> r(a);
125  while(!r.compare_exchange_strong(
126  rd, R::combine(rd, b), std::memory_order_relaxed))
127  ;
128  }
129  }
130  // Legion actually requires this additional interface:
131  static constexpr const T & identity =
132  detail::identity_traits<R>::template value<T>;
133  template<bool E>
134  static void fold(RHS & a, RHS b) {
135  apply<E>(a, b);
136  }
137 
138 private:
139  static void init() {
140  {
141  log::devel_guard guard(reduction_wrapper_tag);
142  flog_devel(info) << "registering reduction operation " << util::type<R>()
143  << " for " << util::type<T>() << std::endl;
144  }
145 
146  // Register the operation with the Legion runtime
147  Legion::Runtime::register_reduction_op<wrap>(id);
148  }
149 
150 public:
151  // NB: 0 is reserved by Legion.
152  static inline const Legion::ReductionOpID id =
153  (run::context::instance().register_init(init), reduction_counter());
154 };
