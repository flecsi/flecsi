236  const point_t & max_b1,
237  const point_t & min_b2,
238  const point_t & max_b2) {
239  return !((max_b1[0] < min_b2[0] || max_b1[1] < min_b2[1] ||
240  max_b1[2] < min_b2[2]) ||
241  (max_b2[0] < min_b1[0] || max_b2[1] < min_b1[1] ||
242  max_b2[2] < min_b1[2]));
243  }
244 
246  static bool intersects_sphere_sphere(const point_t & c1,
247  const element_t r1,
248  const point_t & c2,
249  const element_t r2) {
250  return (c2[0] - c1[0]) * (c2[0] - c1[0]) +
251  (c2[1] - c1[1]) * (c2[1] - c1[1]) +
252  (c2[2] - c1[2]) * (c2[2] - c1[2]) <=
253  (r1 + r2) * (r1 + r2);
254  }

258  static bool intersects_sphere_box(const point_t & min,
259  const point_t & max,
260  const point_t & c,
261  const element_t & r) {
262  point_t x = point_t(c[0] < max[0] ? c[0] : max[0],
263  c[1] < max[1] ? c[1] : max[1],
264  c[2] < max[2] ? c[2] : max[2]);
265  x = {x[0] < min[0] ? min[0] : x[0],
266  x[1] < min[1] ? min[1] : x[1],
267  x[2] < min[2] ? min[2] : x[2]};
268  element_t dist = (x[0] - c[0]) * (x[0] - c[0]) +
269  (x[1] - c[1]) * (x[1] - c[1]) +
270  (x[2] - c[2]) * (x[2] - c[2]);
271  return dist <= r * r;
272  }
