203  /*
204  Using the Mesh Index Space (MIS) ordering, compute intervals,

210  template<PrivilegeCount N>
211  static void idx_itvls(index_coloring const & ic,
212  std::vector<std::size_t> & num_intervals,
213  std::vector<std::pair<std::size_t, std::size_t>> & intervals,
214  std::map<Color, std::vector<std::pair<std::size_t, std::size_t>>> &
215  src_points,
216  field<util::id>::accessor1<privilege_cat<privilege_repeat<wo, N - (N > 1)>,
217  privilege_repeat<na, (N > 1)>>> fmap,
218  std::map<std::size_t, std::size_t> & rmap,
219  MPI_Comm const & comm) {
220  std::vector<std::size_t> entities;
221 
222  /*
223  Define the entity ordering from coloring. This version uses the
224  mesh ordering, i.e., the entities are sorted by ascending mesh id.
225  */
226 
227  for(auto e : ic.owned) {
228  entities.push_back(e);
229  } // for
230 
231  auto [rank, size] = util::mpi::info(comm);
232 
233  std::vector<std::vector<std::size_t>> requests(size);
234  for(auto e : ic.ghosts) {
235  entities.push_back(e.id);
236  requests[e.color].emplace_back(e.id);
237  } // for
238 
239  /*
240  This call is what actually establishes the entity ordering by
241  sorting the mesh entity ids.
242  */
243 
244  util::force_unique(entities);
245 
246  /*
247  Initialize the forward and reverse maps.
248  */
249 
250  flog_assert(entities.size() == (ic.owned.size() + ic.ghosts.size()),
251  "entities size(" << entities.size() << ") doesn't match sum of owned("
252  << ic.owned.size() << ") and ghosts(" << ic.ghosts.size()
253  << ")");
254 
255  std::size_t off{0};
256  rmap.clear();
257  for(auto e : entities) {
258  fmap[off] = e;
259  rmap[e] = off++;
260  }
261 
262  /*
263  After the entity order has been established, we need to create
264  a lookup table for local ghost offsets.
265  */
266 
267  std::map<std::size_t, std::size_t> ghost_offsets;
268  for(auto e : ic.ghosts) {
269  auto it = std::find(entities.begin(), entities.end(), e.id);
270  flog_assert(it != entities.end(), "ghost entity doesn't exist");
271  ghost_offsets[e.id] = std::distance(entities.begin(), it);
272  } // for
273 
274  /*
275  We also need to create a lookup table so that we can provide
276  local shared offset information to other processes that request it.
277  */
278 
279  std::map<std::size_t, std::size_t> shared_offsets;
280  for(auto & e : ic.shared) {
281  auto it = std::find(entities.begin(), entities.end(), e.id);
282  flog_assert(it != entities.end(), "shared entity doesn't exist");
283  shared_offsets[e.id] = std::distance(entities.begin(), it);
284  } // for
285 
286  /*
287  Send/Receive requests for shared offsets with other processes.
288  */
289 
290  auto requested = util::mpi::all_to_allv(
291  [&requests](int r, int) -> auto & { return requests[r]; }, comm);
292 
293  /*
294  Fulfill the requests that we received from other processes, i.e.,
295  provide the locaL offset for the requested mesh ids.
296  */
297 
298  std::vector<std::vector<std::size_t>> fulfills(size);
299  {
300  int r = 0;
301  for(const auto & rv : requested) {
302  for(auto c : rv) {
303  fulfills[r].emplace_back(shared_offsets[c]);
304  } // for
305  ++r;
306  } // for
307  } // scope
308 
309  /*
310  Send/Receive the local offset information with other processes.
311  */
312 
313  auto fulfilled = util::mpi::all_to_allv(
314  [f = std::move(fulfills)](int r, int) { return std::move(f[r]); }, comm);
315  /*
316  Setup source pointers.
317  */
318 
319  int r = 0;
320  for(const auto & rv : fulfilled) {
321  if(r == rank) {
322  ++r;
323  continue;
324  } // if
325 
326  auto & points = src_points[r];
327  points.reserve(rv.size());
328  auto & request = requests[r];
329 
330  std::size_t i{0};
331  for(auto v : rv) {
332  points.emplace_back(std::make_pair(ghost_offsets[request[i]], v));
333  ++i;
334  }
335  ++r;
336  } // for
337 
338  /*
339  Compute local intervals.
340  */
341 
342  auto g = ghost_offsets.begin();
343  std::size_t begin = 0, run = 0;
344  for(; g != ghost_offsets.end(); ++g) {
345  if(!run || g->second != begin + run) {
346  if(run) {
347  intervals.emplace_back(std::make_pair(begin, begin + run));
348  begin = g->second;
349  }
350  run = 1;
351  }
352  else {
353  ++run;
354  }
355  } // for
356 
357  intervals.emplace_back(std::make_pair(begin, begin + run));
358  std::size_t local_itvls = intervals.size();
359 
360  /*
361  Gather global interval sizes.
362  */
363 
364  num_intervals = util::mpi::all_gather(local_itvls, comm);
365  } // idx_itvls
