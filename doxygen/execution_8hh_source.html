171  program_option(const char * section,
172  const char * flag,
173  const char * help,
174  std::initializer_list<initializer_value> values = {},
175  std::function<bool(boost::any const &, std::stringstream & ss)> check =
176  default_check) {
177  auto semantic_ = boost::program_options::value(&value_);
178 
179  bool zero{false}, implicit{false};
180  for(auto const & v : values) {
181  if(v.first == option_default) {
182  semantic_->default_value(v.second);
183  }
184  else if(v.first == option_implicit) {
185  semantic_->implicit_value(v.second);
186  implicit = true;
187  }
188  else if(v.first == option_zero) {
189  semantic_->zero_tokens();
190  zero = true;
191  }
192  else if(v.first == option_multi) {
193  semantic_->multitoken();
194  }
195  else {
196  flog_fatal("invalid value type");
197  } // if
198  } // for
199 
200  if(zero) {
201  flog_assert(implicit, "option_zero specified without option_implicit");
202  } // if
203 
204  auto option =
205  boost::make_shared<boost::program_options::option_description>(
206  flag, semantic_, help);
207 
208  run::context::instance()
209  .descriptions_map()
210  .try_emplace(section, section)
211  .first->second.add(option);
212 
213  std::string sflag(flag);
214  sflag = sflag.substr(0, sflag.find(','));
215 
216  run::context::instance().option_checks().try_emplace(sflag, false, check);
217  } // program_option
