84 template<class... TT>
85 struct param_buffers : private detail::param_buffers {
86  using Tuple = std::tuple<TT...>;
87 
88  param_buffers(Tuple & t, const std::string & nm) : acc(t), nm(nm) {
89  buffer(std::index_sequence_for<TT...>());
90  }
91  // Prevent using temporaries, which is often unsafe:
92  param_buffers(Tuple &, const std::string &&) = delete;
93 
94  ~param_buffers() noexcept(false) {
95  util::annotation::rguard<util::annotation::execute_task_unbind> ann(nm);
96  std::apply(
97  [this](TT &... tt) {
98  (void)this; // to appease Clang
99  (visit(tt), ...);
100  },
101  acc);
102  }

104 private:
