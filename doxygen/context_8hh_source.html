211  auto print_usage = [this](std::string program,
212  auto const & master,
213  auto const & flecsi) {
214  if(process_ == 0) {
215  std::cout << "Usage: " << program << " ";
216 
217  size_t positional_count = positional_desc_.max_total_count();
218  size_t max_label_chars = std::numeric_limits<size_t>::min();
219 
220  for(size_t i{0}; i < positional_count; ++i) {
221  std::cout << "<" << positional_desc_.name_for_position(i) << "> ";
222 
223  const size_t size = positional_desc_.name_for_position(i).size();
224  max_label_chars = size > max_label_chars ? size : max_label_chars;
225  } // for
226 
227  max_label_chars += 2;
228 
229  std::cout << std::endl << std::endl;
230 
231  if(positional_count) {
232  std::cout << "Positional Options:" << std::endl;
233 
234  for(size_t i{0}; i < positional_desc_.max_total_count(); ++i) {
235  auto const & name = positional_desc_.name_for_position(i);
236  auto help = positional_help_.at(name);
237  std::cout << " " << name << " ";
238  std::cout << std::string(max_label_chars - name.size() - 2, ' ');
239 
240  if(help.size() > 78 - max_label_chars) {
241  std::string first = help.substr(
242  0, help.substr(0, 78 - max_label_chars).find_last_of(' '));
243  std::cout << first << std::endl;
244  help =
245  help.substr(first.size() + 1, help.size() - first.size() + 1);
246 
247  while(help.size() > 78 - max_label_chars) {
248  std::string part = help.substr(
249  0, help.substr(0, 78 - max_label_chars).find_last_of(' '));
250  std::cout << std::string(max_label_chars + 1, ' ') << part
251  << std::endl;
252  help = help.substr(part.size() + 1, help.size() - part.size());
253  } // while
254 
255  std::cout << std::string(max_label_chars + 1, ' ') << help
256  << std::endl;
257  }
258  else {
259  std::cout << help << std::endl;
260  } // if
261 
262  } // for
263 
264  std::cout << std::endl;
265  } // if

267  std::cout << master << std::endl;
268  std::cout << flecsi << std::endl;
