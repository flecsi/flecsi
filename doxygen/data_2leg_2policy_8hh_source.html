76 struct unique_handle {
77  unique_handle() = default;
78  unique_handle(T t) : h(t) {}
79  unique_handle(unique_handle && u) noexcept : h(std::exchange(u.h, {})) {}
80  // Prevent erroneous conversion through T constructor:
81  template<class U>
82  unique_handle(unique_handle<U> &&) = delete;
83  ~unique_handle() {
84  if(*this) // empty LogicalRegions, at least, cannot be deleted
85  destroy(h);
86  }
87  unique_handle & operator=(unique_handle u) noexcept {
88  std::swap(h, u.h);
89  return *this;
90  }
91  explicit operator bool() {
92  return h.exists();
93  }
94  operator T() const {
95  return h;
96  }

273  field_id_t fid,
